#Конец ГВ Республика
country_event = {
	id = ate_spain.1
	title = ate_spain.1.t
	desc = ate_spain.1.d
	picture = GFX_report_event_spain_civil_war_soldiers
	
	is_triggered_only = yes
	
	option = {
		name = ate_spain.1.a
		add_political_power = 100
		add_timed_idea = {
			idea = SPA_recovering_from_civil_war
			days = 730
		}
		if = {
			limit = { owns_state = 41 }
			set_capital = 41
		}
		clr_global_flag = ate_spanish_civil_war
		set_global_flag = scw_over
		delete_unit_template_and_units = {
			division_template = "Guarnición Republicana"
		}
	}
}
