country_event = {
	id = parlament.2
	title = parlament.2.t
	desc = parlament.2.d
	picture = GFX_report_event_election_vote
	
	is_triggered_only = yes
	
	hidden = yes

	option = {
		name = parlament.2.a
		if = {
			limit = { 
				check_variable = { vote_yes > vote_no }
			}
			ROOT = { country_event = { id = parlament.3 days = 1 } }
		}
		if = {
			limit = {
				OR = {
					check_variable = { vote_no > vote_yes }
					check_variable = { vote_no = vote_yes }
				}
			}
			ROOT = { country_event = { id = parlament.4 days = 1 } }
		}
		clear_variable = vote_yes
		clear_variable = vote_no
		clr_country_flag = ATE_PAR_dem_yes
		clr_country_flag = ATE_PAR_dem_no
		clr_country_flag = ATE_PAR_lib_yes
		clr_country_flag = ATE_PAR_lib_no
		clr_country_flag = ATE_PAR_soc_yes
		clr_country_flag = ATE_PAR_soc_no
		clr_country_flag = ATE_PAR_fas_yes
		clr_country_flag = ATE_PAR_fas_no
		clr_country_flag = ATE_PAR_cent_yes
		clr_country_flag = ATE_PAR_cent_no
		clr_country_flag = ATE_PAR_com_yes
		clr_country_flag = ATE_PAR_com_no
		clr_country_flag = ATE_PAR_neut_yes
		clr_country_flag = ATE_PAR_neut_no
	}
}
